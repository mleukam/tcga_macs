filepaths <- map(tcga_manifest, function(df){
  df %>% dplyr::select(filename) %>% 
    mutate(filename = (str_replace(filename, ".gz", ""))) %>%
    mutate(filepath = (str_c("data/htseq_counts/", filename)))
           })
